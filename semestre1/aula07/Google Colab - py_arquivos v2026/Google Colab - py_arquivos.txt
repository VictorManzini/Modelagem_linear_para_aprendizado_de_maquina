# O comando open é usado para abrir arquivos.
# O comando with é usado para garantir a finalização do arquivo depois de aberto de maneira segura.
# Quando um arquivo é aberto, ele pode ser aberto para leitura ('r'), escrita ('w'),
# anexar (acrescentar) ('a') ou ler como binário ('rb').
# Para mais informações, acesse: https://docs.python.org/3/library/io.html

with open('/content/drive/MyDrive/Colab Notebooks/texto_utf8.txt', 'r') as file:
   print(file)               # exibe o que foi carregado na variável file ao abrir o arquivo
   print(type(file))         # exibe o tipo de dado (estrutura) da variável file
   linhas = file.readlines() # readlines é um método da classe _io.TextIOWrapper que lê as linhas do arquivo
   print(linhas)             # as linhas são armazenadas em uma lista (vetor) de strings

import librosa                   # processamento de áudio
import librosa.display           # plot de gráficos
import matplotlib.pyplot as plt  # plot de gráficos

import IPython                   # para tocar áudio no Jupyter

# carregando o arquivo de áudio
sinal, sr = librosa.load('/content/drive/MyDrive/Colab Notebooks/aeiou_ruido.wav', sr=44100)

librosa.display.waveshow(sinal, sr=sr,color='r')
plt.grid(color='k', linestyle='-', linewidth=0.3)
plt.xlabel('Time (s)', fontsize=14)
plt.ylabel('Amplitude (a.u.)', fontsize=14)
